Language Comparision

Ruby

values = { 'foo' => 1, 'bar' => 2, 'baz' => 3 }
values.each do |k, v|
  puts "#{k}: #{v}"
end

foo: 1
bar: 2
baz: 3

Python

values = { 'foo': 1, 'bar': 2, 'baz': 3 }
for k in values:
  print k + ': ' + `values[k]`
print '----'
for k, v in values.items():
  print k + ': ' + `v`

baz: 3
foo: 1
bar: 2
----
baz: 3
foo: 1
bar: 2

Perl

use feature say;

%values = (foo => 1, bar => 2, baz => 3);
while (my ($k, $v) = each(%values)) {
  say "$k: $v";
}
say '----';
say "$_: $values{$_}" for keys(%values);

foo: 1
bar: 2
baz: 3
----
foo: 1
bar: 2
baz: 3

PHP

<?php
$values = array('foo' => 1, 'bar' => 2, 'baz' => 3);
foreach ($values as $k => $v) {
  echo "$k: $v\n";
}

foo: 1
bar: 2
baz: 3
