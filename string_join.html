Language Comparision

Ruby

words = %w(foo bar baz)
puts words.join(' - ')

foo - bar - baz

Python

words = ['foo', 'bar', 'baz']
print ' - '.join(words)

foo - bar - baz

Perl

use feature say;

@words = qw(foo bar baz);
say join(' - ', @words);

foo - bar - baz

PHP

<?php
$words = array('foo', 'bar', 'baz');
echo implode(' - ', $words), "\n";


foo - bar - baz
