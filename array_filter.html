Language Comparision

Python

nums = [1, 2, 3, 4, 5]
odds = [x for x in nums if x % 2 != 0 ]
print odds

[1, 3, 5]

Perl

require 'utils/util.pm';

@nums = (1, 2, 3, 4, 5);
foreach $n (@nums) {
  push @odds, $n if $n % 2 != 0;
}
print_array(@odds);

[1, 3, 5]

PHP

<?php
require('utils/util.php');

$nums = array(1, 2, 3, 4, 5);
$odds = array_filter($nums, function($x) { return $x % 2 != 0; });
print_array($odds);

[1, 3, 5]
